}



/*PROCESSOR DIRECTIVE AND DECLARATION OF NAMESPACE
#include <iostream> // preprocessor directive - include iostream file  
using namespace std; // declare use of the standard namespace

int main(){
cout << "Hello World" << endl;

return 0;
}

*/

/* DECLARING WITHOUT NAMESPACE 
#include <iostream> // preprocessor directive - include iostream file

// std:: is required if declaring using namespace std at the  beginning
std::cout << "Hello World" << std::endl;  return 0;
}
